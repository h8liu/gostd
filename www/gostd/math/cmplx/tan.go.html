// Complex circular tangent
//
// DESCRIPTION:
//
35// If
//     z = x + iy,
//
// then
//
40//           sin 2x  +  i sinh 2y
//     w  =  --------------------.
//            cos 2x  +  cosh 2y
//
// On the real axis the denominator is zero at odd multiples
45// of PI/2.  The denominator is evaluated by its Taylor
// series near these points.
//
// ctan(z) = -i ctanh(iz).
//
50// ACCURACY:
//
//                      Relative error:
// arithmetic   domain     # trials      peak         rms
//    DEC       -10,+10      5200       7.1e-17     1.6e-17
55//    IEEE      -10,+10     30000       7.2e-16     1.2e-16
// Also tested by ctan * ccot = 1 and catan(ctan(z))  =  z.

// Tan returns the tangent of x.
func Tan(x complex128) complex128 {
60    d := math.Cos(2*real(x)) + math.Cosh(2*imag(x))
    if math.Abs(d) < 0.25 {
        d = tanSeries(x)
    }
    if d == 0 {
65        return Inf()
    }
    return complex(math.Sin(2*real(x))/d, math.Sinh(2*imag(x))/d)
}

100    if t >= 0 {

110func tanSeries(z complex128) float64 {

    y := math.Abs(2 * imag(z))
    x = reducePi(x)
115    x = x * x
    y = y * y
    x2 := 1.0
    y2 := 1.0
    f := 1.0
120    rn := 0.0
    d := 0.0
    for {

        f *= rn
125        rn += 1
        f *= rn

        y2 *= y
        t := y2 + x2
130        t /= f

        f *= rn
135        rn += 1

        t = y2 - x2

// Complex circular cotangent
150//
// DESCRIPTION:
//
// If
//     z = x + iy,
155//
// then
//
//           sin 2x  -  i sinh 2y
//     w  =  --------------------.

// multiples of PI/2.  Near these points it is evaluated
// by a Taylor series.
165//
// ACCURACY:
//
//                      Relative error:
// arithmetic   domain     # trials      peak         rms
170//    DEC       -10,+10      3000       6.5e-17     1.6e-17
//    IEEE      -10,+10     30000       9.2e-16     1.2e-16
// Also tested by ctan * ccot = 1 + i0.

// Cot returns the cotangent of x.
175func Cot(x complex128) complex128 {
    d := math.Cosh(2*imag(x)) - math.Cos(2*real(x))
    if math.Abs(d) < 0.25 {
        d = tanSeries(x)
    }
180    if d == 0 {
        return Inf()
    }
    return complex(math.Sin(2*real(x))/d, -math.Sinh(2*imag(x))/d)
}

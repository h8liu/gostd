        key, err1 := QueryUnescape(key)
        if err1 != nil {
            if err == nil {
                err = err1
565            }
            continue
        }
        value, err1 = QueryUnescape(value)
        if err1 != nil {
570            if err == nil {
                err = err1
            }
            continue
        }
575        m[key] = append(m[key], value)
    }
    return err
}

580// Encode encodes the values into ``URL encoded'' form
// ("bar=baz&foo=quux") sorted by key.
func (v Values) Encode() string {
    if v == nil {
        return ""
585    }
    var buf bytes.Buffer
    keys := make([]string, 0, len(v))
    for k := range v {
        keys = append(keys, k)
590    }
    sort.Strings(keys)
    for _, k := range keys {
        vs := v[k]
        prefix := QueryEscape(k) + "="
595        for _, v := range vs {
            if buf.Len() > 0 {
                buf.WriteByte('&')
            }
            buf.WriteString(prefix)
600            buf.WriteString(QueryEscape(v))
        }
    }
    return buf.String()
}

// resolvePath applies special path segments from refs and applies
// them to base, per RFC 3986.
func resolvePath(base, ref string) string {
    var full string
610    if ref == "" {
        full = base
    } else if ref[0] != '/' {
        i := strings.LastIndex(base, "/")
        full = base[:i+1] + ref
615    } else {
        full = ref
    }
    if full == "" {
        return ""
620    }
    var dst []string
    src := strings.Split(full, "/")
    for _, elem := range src {
        switch elem {
625        case ".":
            // drop
        case "..":
            if len(dst) > 0 {
                dst = dst[:len(dst)-1]
630            }
        default:
            dst = append(dst, elem)
        }
    }
635    if last := src[len(src)-1]; last == "." || last == ".." {
        // Add final slash to the joined path.
        dst = append(dst, "")
    }
    return "/" + strings.TrimLeft(strings.Join(dst, "/"), "/")
640}

// IsAbs returns true if the URL is absolute.
func (u *URL) IsAbs() bool {
    return u.Scheme != ""
645}

// Parse parses a URL in the context of the receiver.  The provided URL
// may be relative or absolute.  Parse returns nil, err on parse
// failure, otherwise its return value is the same as ResolveReference.
650func (u *URL) Parse(ref string) (*URL, error) {
    refurl, err := Parse(ref)
    if err != nil {
        return nil, err
    }
655    return u.ResolveReference(refurl), nil
}

// ResolveReference resolves a URI reference to an absolute URI from
// an absolute base URI, per RFC 3986 Section 5.2.  The URI reference
660// may be relative or absolute.  ResolveReference always returns a new
// URL instance, even if the returned URL is identical to either the
// base or reference. If ref is an absolute URL, then ResolveReference
// ignores base and returns a copy of ref.
func (u *URL) ResolveReference(ref *URL) *URL {
665    url := *ref
    if ref.Scheme == "" {
        url.Scheme = u.Scheme
    }
    if ref.Scheme != "" || ref.Host != "" || ref.User != nil {
670        // The "absoluteURI" or "net_path" cases.
        url.Path = resolvePath(ref.Path, "")
        return &url
    }
    if ref.Opaque != "" {
675        url.User = nil
        url.Host = ""
        url.Path = ""
        return &url
    }
680    if ref.Path == "" {
        if ref.RawQuery == "" {
            url.RawQuery = u.RawQuery
            if ref.Fragment == "" {
                url.Fragment = u.Fragment
685            }
        }
    }
    // The "abs_path" or "rel_path" cases.
    url.Host = u.Host
690    url.User = u.User
    url.Path = resolvePath(u.Path, ref.Path)
    return &url
}

695// Query parses RawQuery and returns the corresponding values.

    v, _ := ParseQuery(u.RawQuery)
    return v
}

// RequestURI returns the encoded path?query or opaque?query
// string that would be used in an HTTP request for u.
func (u *URL) RequestURI() string {
    result := u.Opaque
705    if result == "" {
        result = escape(u.Path, encodePath)
        if result == "" {
            result = "/"
        }
710    } else {
        if strings.HasPrefix(result, "//") {
            result = u.Scheme + ":" + result
        }
    }
715    if u.RawQuery != "" {
        result += "?" + u.RawQuery
    }
    return result
}

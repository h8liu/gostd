func sng(w io.WriteCloser, filename string, png image.Image) {
    defer w.Close()
    bounds := png.Bounds()
    cm := png.ColorModel()
70    var bitdepth int
    switch cm {
    case color.RGBAModel, color.NRGBAModel, color.AlphaModel, color.GrayModel:
        bitdepth = 8
    default:
75        bitdepth = 16
    }
    cpm, _ := cm.(color.Palette)
    var paletted *image.Paletted
    if cpm != nil {
80        switch {
        case len(cpm) <= 2:
            bitdepth = 1
        case len(cpm) <= 4:
            bitdepth = 2
85        case len(cpm) <= 16:
            bitdepth = 4
        default:
            bitdepth = 8
        }
90        paletted = png.(*image.Paletted)
    }

    // Write the filename and IHDR.
    io.WriteString(w, "#SNG: from "+filename+".png\nIHDR {\n")
95    fmt.Fprintf(w, "    width: %d; height: %d; bitdepth: %d;\n", bounds.Dx(), bounds.Dy(), bitdepth)
    switch {
    case cm == color.RGBAModel, cm == color.RGBA64Model:
        io.WriteString(w, "    using color;\n")
    case cm == color.NRGBAModel, cm == color.NRGBA64Model:
100        io.WriteString(w, "    using color alpha;\n")
    case cm == color.GrayModel, cm == color.Gray16Model:
        io.WriteString(w, "    using grayscale;\n")
    case cpm != nil:
        io.WriteString(w, "    using color palette;\n")
105    default:
        io.WriteString(w, "unknown PNG decoder color model\n")
    }
    io.WriteString(w, "}\n")

110    // We fake a gAMA output. The test files have a gAMA chunk but the go PNG parser ignores it
    // (the PNG spec section 11.3 says "Ancillary chunks may be ignored by a decoder").
    io.WriteString(w, "gAMA {1.0000}\n")

    // Write the PLTE and tRNS (if applicable).
115    if cpm != nil {
        lastAlpha := -1
        io.WriteString(w, "PLTE {\n")
        for i, c := range cpm {
            var r, g, b, a uint8
120            switch c := c.(type) {
            case color.RGBA:
                r, g, b, a = c.R, c.G, c.B, 0xff
            case color.NRGBA:
                r, g, b, a = c.R, c.G, c.B, c.A
125            default:
                panic("unknown palette color type")
            }
            if a != 0xff {
                lastAlpha = i
130            }
            fmt.Fprintf(w, "    (%3d,%3d,%3d)     # rgb = (0x%02x,0x%02x,0x%02x)\n", r, g, b, r, g, b)
        }
        io.WriteString(w, "}\n")
        if lastAlpha != -1 {
135            io.WriteString(w, "tRNS {\n")
            for i := 0; i <= lastAlpha; i++ {
                _, _, _, a := cpm[i].RGBA()
                a >>= 8
                fmt.Fprintf(w, " %d", a)
140            }
            io.WriteString(w, "}\n")
        }
    }

145    // Write the IMAGE.
    io.WriteString(w, "IMAGE {\n    pixels hex\n")
    for y := bounds.Min.Y; y < bounds.Max.Y; y++ {
        switch {
        case cm == color.GrayModel:
150            for x := bounds.Min.X; x < bounds.Max.X; x++ {
                gray := png.At(x, y).(color.Gray)
                fmt.Fprintf(w, "%02x", gray.Y)
            }
        case cm == color.Gray16Model:
155            for x := bounds.Min.X; x < bounds.Max.X; x++ {
                gray16 := png.At(x, y).(color.Gray16)
                fmt.Fprintf(w, "%04x ", gray16.Y)
            }
        case cm == color.RGBAModel:
160            for x := bounds.Min.X; x < bounds.Max.X; x++ {
                rgba := png.At(x, y).(color.RGBA)
                fmt.Fprintf(w, "%02x%02x%02x ", rgba.R, rgba.G, rgba.B)
            }
        case cm == color.RGBA64Model:
165            for x := bounds.Min.X; x < bounds.Max.X; x++ {
                rgba64 := png.At(x, y).(color.RGBA64)
                fmt.Fprintf(w, "%04x%04x%04x ", rgba64.R, rgba64.G, rgba64.B)
            }
        case cm == color.NRGBAModel:
170            for x := bounds.Min.X; x < bounds.Max.X; x++ {
                nrgba := png.At(x, y).(color.NRGBA)
                fmt.Fprintf(w, "%02x%02x%02x%02x ", nrgba.R, nrgba.G, nrgba.B, nrgba.A)
            }
        case cm == color.NRGBA64Model:
175            for x := bounds.Min.X; x < bounds.Max.X; x++ {
                nrgba64 := png.At(x, y).(color.NRGBA64)
                fmt.Fprintf(w, "%04x%04x%04x%04x ", nrgba64.R, nrgba64.G, nrgba64.B, nrgba64.A)
            }
        case cpm != nil:
180            var b, c int
            for x := bounds.Min.X; x < bounds.Max.X; x++ {
                b = b<<uint(bitdepth) | int(paletted.ColorIndexAt(x, y))
                c++
                if c == 8/bitdepth {
185                    fmt.Fprintf(w, "%02x", b)
                    b = 0
                    c = 0
                }
            }
190            if c != 0 {
                for c != 8/bitdepth {
                    b = b << uint(bitdepth)
                    c++
                }
195                fmt.Fprintf(w, "%02x", b)
            }
        }
        io.WriteString(w, "\n")
    }
200    io.WriteString(w, "}\n")
}

func TestReader(t *testing.T) {
    names := filenames
205    if testing.Short() {

210        img, err := readPNG("testdata/pngsuite/" + fn + ".png")
        if err != nil {
            t.Error(fn, err)
            continue
        }
215
        if fn == "basn4a16" {
            // basn4a16.sng is gray + alpha but sng() will produce true color + alpha
            // so we just check a single random pixel.
            c := img.At(2, 1).(color.NRGBA64)
220            if c.R != 0x11a7 || c.G != 0x11a7 || c.B != 0x11a7 || c.A != 0x1085 {

            }
            continue
        }
225
        piper, pipew := io.Pipe()
        pb := bufio.NewScanner(piper)
        go sng(pipew, fn, img)
        defer piper.Close()
230
        // Read the .sng file.
        sf, err := os.Open("testdata/pngsuite/" + fn + ".sng")
        if err != nil {
            t.Error(fn, err)
235            continue
        }
        defer sf.Close()
        sb := bufio.NewScanner(sf)
        if err != nil {
240            t.Error(fn, err)
            continue
        }

        // Compare the two, in SNG format, line by line.
245        for {
            pdone := !pb.Scan()
            sdone := !sb.Scan()
            if pdone && sdone {
                break
250            }
            if pdone || sdone {
                t.Errorf("%s: Different sizes", fn)
                break
            }
255            ps := pb.Text()
            ss := sb.Text()
            if ps != ss {
                t.Errorf("%s: Mismatch\n%sversus\n%s\n", fn, ps, ss)
                break
260            }
        }
        if pb.Err() != nil {
            t.Error(fn, pb.Err())
        }
265        if sb.Err() != nil {
            t.Error(fn, sb.Err())
        }
    }

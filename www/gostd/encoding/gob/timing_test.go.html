func (b *benchmarkBuf) Read(p []byte) (n int, err error) {
210    n = copy(p, b.data[b.offset:])
    if n == 0 {
        return 0, io.EOF
    }
    b.offset += n
215    return
}

func (b *benchmarkBuf) ReadByte() (c byte, err error) {
    if b.offset >= len(b.data) {
220        return 0, io.EOF
    }
    c = b.data[b.offset]
    b.offset++
    return
225}

func (b *benchmarkBuf) reset() {
    b.offset = 0

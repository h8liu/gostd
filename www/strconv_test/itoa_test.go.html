func TestItoa(t *testing.T) {
    for _, test := range itob64tests {
        s := FormatInt(test.in, test.base)
        if s != test.out {
65            t.Errorf("FormatInt(%v, %v) = %v want %v",
                test.in, test.base, s, test.out)
        }
        x := AppendInt([]byte("abc"), test.in, test.base)
        if string(x) != "abc"+test.out {
70            t.Errorf("AppendInt(%q, %v, %v) = %q want %v",
                "abc", test.in, test.base, x, test.out)
        }

        if test.in >= 0 {
75            s := FormatUint(uint64(test.in), test.base)
            if s != test.out {
                t.Errorf("FormatUint(%v, %v) = %v want %v",
                    test.in, test.base, s, test.out)
            }
80            x := AppendUint(nil, uint64(test.in), test.base)

// has the corresponding value.

//

595// set with Init. Token positions are relative to that file

600    s.skipWhitespace()

    // current token start
    pos = s.file.Pos(s.offset)

605    // determine token value
    insertSemi := false
    switch ch := s.ch; {
    case isLetter(ch):
        lit = s.scanIdentifier()
610        if len(lit) > 1 {
            // keywords are longer than one letter - avoid lookup otherwise
            tok = token.Lookup(lit)
            switch tok {
            case token.IDENT, token.BREAK, token.CONTINUE, token.FALLTHROUGH, token.RETURN:
615                insertSemi = true
            }
        } else {
            insertSemi = true
            tok = token.IDENT
620        }
    case '0' <= ch && ch <= '9':
        insertSemi = true
        tok, lit = s.scanNumber(false)
    default:
625        s.next() // always make progress
        switch ch {
        case -1:
            if s.insertSemi {
                s.insertSemi = false // EOF consumed
630                return pos, token.SEMICOLON, "\n"
            }
            tok = token.EOF
        case '\n':
            // we only reach here if s.insertSemi was
635            // set in the first place and exited early
            // from s.skipWhitespace()
            s.insertSemi = false // newline consumed
            return pos, token.SEMICOLON, "\n"
        case '"':
640            insertSemi = true
            tok = token.STRING
            lit = s.scanString()
        case '\'':
            insertSemi = true
645            tok = token.CHAR
            lit = s.scanRune()
        case '`':
            insertSemi = true
            tok = token.STRING
650            lit = s.scanRawString()
        case ':':
            tok = s.switch2(token.COLON, token.DEFINE)
        case '.':
            if '0' <= s.ch && s.ch <= '9' {
655                insertSemi = true
                tok, lit = s.scanNumber(true)
            } else if s.ch == '.' {
                s.next()
                if s.ch == '.' {
660                    s.next()
                    tok = token.ELLIPSIS
                }
            } else {
                tok = token.PERIOD
665            }
        case ',':
            tok = token.COMMA
        case ';':
            tok = token.SEMICOLON
670            lit = ";"
        case '(':
            tok = token.LPAREN
        case ')':
            insertSemi = true
675            tok = token.RPAREN
        case '[':
            tok = token.LBRACK
        case ']':
            insertSemi = true
680            tok = token.RBRACK
        case '{':
            tok = token.LBRACE
        case '}':
            insertSemi = true
685            tok = token.RBRACE
        case '+':
            tok = s.switch3(token.ADD, token.ADD_ASSIGN, '+', token.INC)
            if tok == token.INC {
                insertSemi = true
690            }
        case '-':
            tok = s.switch3(token.SUB, token.SUB_ASSIGN, '-', token.DEC)
            if tok == token.DEC {
                insertSemi = true
695            }
        case '*':
            tok = s.switch2(token.MUL, token.MUL_ASSIGN)
        case '/':
            if s.ch == '/' || s.ch == '*' {
700                // comment
                if s.insertSemi && s.findLineEnd() {
                    // reset position to the beginning of the comment
                    s.ch = '/'
                    s.offset = s.file.Offset(pos)
705                    s.rdOffset = s.offset + 1
                    s.insertSemi = false // newline consumed
                    return pos, token.SEMICOLON, "\n"
                }
                lit = s.scanComment()
710                if s.mode&ScanComments == 0 {
                    // skip comment
                    s.insertSemi = false // newline consumed
                    goto scanAgain
                }
715                tok = token.COMMENT
            } else {
                tok = s.switch2(token.QUO, token.QUO_ASSIGN)
            }
        case '%':
720            tok = s.switch2(token.REM, token.REM_ASSIGN)
        case '^':
            tok = s.switch2(token.XOR, token.XOR_ASSIGN)
        case '<':
            if s.ch == '-' {
725                s.next()
                tok = token.ARROW
            } else {
                tok = s.switch4(token.LSS, token.LEQ, '<', token.SHL, token.SHL_ASSIGN)
            }
730        case '>':
            tok = s.switch4(token.GTR, token.GEQ, '>', token.SHR, token.SHR_ASSIGN)
        case '=':
            tok = s.switch2(token.ASSIGN, token.EQL)
        case '!':
735            tok = s.switch2(token.NOT, token.NEQ)
        case '&':
            if s.ch == '^' {
                s.next()
                tok = s.switch2(token.AND_NOT, token.AND_NOT_ASSIGN)
740            } else {
                tok = s.switch3(token.AND, token.AND_ASSIGN, '&', token.LAND)
            }
        case '|':
            tok = s.switch3(token.OR, token.OR_ASSIGN, '|', token.LOR)
745        default:
            // next reports unexpected BOMs - don't repeat
            if ch != bom {
                s.error(s.file.Offset(pos), fmt.Sprintf("illegal character %#U", ch))
            }
750            insertSemi = s.insertSemi // preserve insertSemi info
            tok = token.ILLEGAL
            lit = string(ch)
        }
    }
755    if s.mode&dontInsertSemis == 0 {
        s.insertSemi = insertSemi
    }

    return
760}

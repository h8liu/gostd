            "continued\r\n"+

50            "\r\n")), nil)
    if err != nil {
        t.Fatal(err)
    }
    want := "Body here\ncontinued"
55    buf := make([]byte, len(want))
    n, err := res.Body.Read(buf)
    if n != len(want) || err != io.EOF {
        t.Logf("body = %#v", res.Body)
        t.Errorf("Read = %v, %v; want %d, EOF", n, err, len(want))
60    }
    if string(buf) != want {
        t.Errorf("buf = %q; want %q", buf, want)
    }
}

var setenvEinvalTests = []struct {
15    k, v string
}{
    {"", ""},      // empty key
    {"k=v", ""},   // '=' in key
    {"\x00", ""},  // '\x00' in key
20    {"k", "\x00"}, // '\x00' in value
}

func TestSetenvUnixEinval(t *testing.T) {
    for _, tt := range setenvEinvalTests {
25        err := Setenv(tt.k, tt.v)
        if err == nil {
            t.Errorf(`Setenv(%q, %q) == nil, want error`, tt.k, tt.v)
        }
    }
30}

func TestHMAC(t *testing.T) {
    for i, tt := range hmacTests {
525        h := New(tt.hash, tt.key)
        if s := h.Size(); s != tt.size {
            t.Errorf("Size: got %v, want %v", s, tt.size)
        }
        if b := h.BlockSize(); b != tt.blocksize {
530            t.Errorf("BlockSize: got %v, want %v", b, tt.blocksize)
        }
        for j := 0; j < 2; j++ {
            n, err := h.Write(tt.in)
            if n != len(tt.in) || err != nil {
535                t.Errorf("test %d.%d: Write(%d) = %d, %v", i, j, len(tt.in), n, err)
                continue
            }

            // Repetitive Sum() calls should return the same value
540            for k := 0; k < 2; k++ {
                sum := fmt.Sprintf("%x", h.Sum(nil))
                if sum != tt.out {
                    t.Errorf("test %d.%d.%d: have %s want %s\n", i, j, k, sum, tt.out)
                }
545            }

            // Second iteration: make sure reset works.
            h.Reset()
        }
550    }
}

20func ExampleRel() {
    paths := []string{
        "/a/b/c",
        "/b/c",
        "./b/c",
25    }
    base := "/a"

    fmt.Println("On Unix:")
    for _, p := range paths {
30        rel, err := filepath.Rel(base, p)
        fmt.Printf("%q: %q %v\n", p, rel, err)
    }

    // Output:
35    // On Unix:
    // "/a/b/c": "b/c" <nil>
    // "/b/c": "../b/c" <nil>
    // "./b/c": "" Rel: can't make b/c relative to /a
}

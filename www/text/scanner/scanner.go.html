// message to os.Stderr.
550func (s *Scanner) Scan() rune {
    ch := s.Peek()

    // reset token text position
    s.tokPos = -1
555    s.Line = 0

redo:
    // skip white space
    for s.Whitespace&(1<<uint(ch)) != 0 {
560        ch = s.next()
    }

    // start collecting token text
    s.tokBuf.Reset()
565    s.tokPos = s.srcPos - s.lastCharLen

    // set token position
    // (this is a slightly optimized version of the code in Pos())
    s.Offset = s.srcBufOffset + s.tokPos
570    if s.column > 0 {
        // common case: last character was not a '\n'
        s.Line = s.line
        s.Column = s.column
    } else {
575        // last character was a '\n'
        // (we cannot be at the beginning of the source
        // since we have called next() at least once)
        s.Line = s.line - 1
        s.Column = s.lastLineLen
580    }

    // determine token value
    tok := ch
    switch {
585    case s.isIdentRune(ch, 0):
        if s.Mode&ScanIdents != 0 {
            tok = Ident
            ch = s.scanIdentifier()
        } else {
590            ch = s.next()
        }
    case isDecimal(ch):
        if s.Mode&(ScanInts|ScanFloats) != 0 {
            tok, ch = s.scanNumber(ch)
595        } else {
            ch = s.next()
        }
    default:
        switch ch {
600        case '"':
            if s.Mode&ScanStrings != 0 {
                s.scanString('"')
                tok = String
            }
605            ch = s.next()
        case '\'':
            if s.Mode&ScanChars != 0 {
                s.scanChar()
                tok = Char
610            }
            ch = s.next()
        case '.':
            ch = s.next()
            if isDecimal(ch) && s.Mode&ScanFloats != 0 {
615                tok = Float
                ch = s.scanMantissa(ch)
                ch = s.scanExponent(ch)
            }
        case '/':
620            ch = s.next()
            if (ch == '/' || ch == '*') && s.Mode&ScanComments != 0 {
                if s.Mode&SkipComments != 0 {
                    s.tokPos = -1 // don't collect token text
                    ch = s.scanComment(ch)
625                    goto redo
                }
                ch = s.scanComment(ch)
                tok = Comment
            }
630        case '`':
            if s.Mode&ScanRawStrings != 0 {
                s.scanRawString()
                tok = String
            }
635            ch = s.next()
        default:
            ch = s.next()
        }
    }
640
    // end of token text
    s.tokEnd = s.srcPos - s.lastCharLen

    s.ch = ch
645    return tok
}

// Pos returns the position of the character immediately after
// the character or token returned by the last call to Next or Scan.
650func (s *Scanner) Pos() (pos Position) {
    pos.Filename = s.Filename
    pos.Offset = s.srcBufOffset + s.srcPos - s.lastCharLen
    switch {
    case s.column > 0:
655        // common case: last character was not a '\n'
        pos.Line = s.line
        pos.Column = s.column
    case s.lastLineLen > 0:
        // last character was a '\n'
660        pos.Line = s.line - 1
        pos.Column = s.lastLineLen
    default:
        // at the beginning of the source
        pos.Line = 1
665        pos.Column = 1
    }
    return
}

670// TokenText returns the string corresponding to the most recently scanned token.
// Valid after calling Scan().
func (s *Scanner) TokenText() string {
    if s.tokPos < 0 {
        // no token text
675        return ""
    }

    if s.tokEnd < 0 {
        // if EOF was reached, s.tokEnd is set to -1 (s.srcPos == 0)
680        s.tokEnd = s.tokPos
    }

    if s.tokBuf.Len() == 0 {
        // common case: the entire token text is still in srcBuf
685        return string(s.srcBuf[s.tokPos:s.tokEnd])
    }

    // part of the token text was saved in tokBuf: save the rest in
    // tokBuf as well and return its content
690    s.tokBuf.Write(s.srcBuf[s.tokPos:s.tokEnd])
    s.tokPos = s.tokEnd // ensure idempotency of TokenText() call
    return s.tokBuf.String()
}

5// Deep equality test via reflection

package reflect

// During deepValueEqual, must keep track of checks that are
10// in progress.  The comparison algorithm assumes that all
// checks in progress are true when it reencounters them.
// Visited comparisons are stored in a map indexed by visit.
type visit struct {
    a1  uintptr
15    a2  uintptr
    typ Type
}

// Tests for deep equality using reflected types. The map argument tracks
20// comparisons that have already been seen, which allows short circuiting on
// recursive types.
func deepValueEqual(v1, v2 Value, visited map[visit]bool, depth int) bool {
    if !v1.IsValid() || !v2.IsValid() {
        return v1.IsValid() == v2.IsValid()
25    }
    if v1.Type() != v2.Type() {
        return false
    }

30    // if depth > 10 { panic("deepValueEqual") }    // for debugging
    hard := func(k Kind) bool {
        switch k {
        case Array, Map, Slice, Struct:
            return true
35        }
        return false
    }

    if v1.CanAddr() && v2.CanAddr() && hard(v1.Kind()) {
40        addr1 := v1.UnsafeAddr()
        addr2 := v2.UnsafeAddr()
        if addr1 > addr2 {
            // Canonicalize order to reduce number of entries in visited.
            addr1, addr2 = addr2, addr1
45        }

        // Short circuit if references are identical ...
        if addr1 == addr2 {
            return true
50        }

        // ... or already seen
        typ := v1.Type()
        v := visit{addr1, addr2, typ}
55        if visited[v] {
            return true
        }

        // Remember for later.
60        visited[v] = true
    }

    switch v1.Kind() {
    case Array:
65        for i := 0; i < v1.Len(); i++ {
            if !deepValueEqual(v1.Index(i), v2.Index(i), visited, depth+1) {
                return false
            }
        }
70        return true
    case Slice:
        if v1.IsNil() != v2.IsNil() {
            return false
        }
75        if v1.Len() != v2.Len() {
            return false
        }
        if v1.Pointer() == v2.Pointer() {
            return true
80        }
        for i := 0; i < v1.Len(); i++ {
            if !deepValueEqual(v1.Index(i), v2.Index(i), visited, depth+1) {
                return false
            }
85        }
        return true
    case Interface:
        if v1.IsNil() || v2.IsNil() {
            return v1.IsNil() == v2.IsNil()
90        }
        return deepValueEqual(v1.Elem(), v2.Elem(), visited, depth+1)
    case Ptr:
        return deepValueEqual(v1.Elem(), v2.Elem(), visited, depth+1)
    case Struct:
95        for i, n := 0, v1.NumField(); i < n; i++ {
            if !deepValueEqual(v1.Field(i), v2.Field(i), visited, depth+1) {
                return false
            }
        }
100        return true
    case Map:
        if v1.IsNil() != v2.IsNil() {
            return false
        }
105        if v1.Len() != v2.Len() {
            return false
        }
        if v1.Pointer() == v2.Pointer() {
            return true
110        }
        for _, k := range v1.MapKeys() {
            if !deepValueEqual(v1.MapIndex(k), v2.MapIndex(k), visited, depth+1) {
                return false
            }
115        }
        return true
    case Func:
        if v1.IsNil() && v2.IsNil() {
            return true
120        }
        // Can't do better than this:
        return false
    default:
        // Normal equality suffices
125        return valueInterface(v1, false) == valueInterface(v2, false)
    }
}

// DeepEqual tests for deep equality. It uses normal == equality where
130// possible but will scan elements of arrays, slices, maps, and fields of
// structs. In maps, keys are compared with == but elements use deep
// equality. DeepEqual correctly handles recursive types. Functions are equal
// only if they are both nil.
// An empty slice is not equal to a nil slice.
135func DeepEqual(a1, a2 interface{}) bool {
    if a1 == nil || a2 == nil {
        return a1 == a2
    }
    v1 := ValueOf(a1)
140    v2 := ValueOf(a2)
    if v1.Type() != v2.Type() {
        return false
    }
    return deepValueEqual(v1, v2, make(map[visit]bool), 0)
145}

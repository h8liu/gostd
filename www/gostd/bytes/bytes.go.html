func IndexFunc(s []byte, f func(r rune) bool) int {
    return indexFunc(s, f, true)
515}

// LastIndexFunc interprets s as a sequence of UTF-8-encoded Unicode code points.
// It returns the byte index in s of the last Unicode
// code point satisfying f(c), or -1 if none do.
520func LastIndexFunc(s []byte, f func(r rune) bool) int {
    return lastIndexFunc(s, f, true)
}

// indexFunc is the same as IndexFunc except that if
525// truth==false, the sense of the predicate function is
// inverted.
func indexFunc(s []byte, f func(r rune) bool, truth bool) int {
    start := 0
    for start < len(s) {
530        wid := 1
        r := rune(s[start])
        if r >= utf8.RuneSelf {
            r, wid = utf8.DecodeRune(s[start:])
        }
535        if f(r) == truth {
            return start
        }
        start += wid
    }
540    return -1
}

// lastIndexFunc is the same as LastIndexFunc except that if
// truth==false, the sense of the predicate function is
545// inverted.
func lastIndexFunc(s []byte, f func(r rune) bool, truth bool) int {
    for i := len(s); i > 0; {
        r, size := rune(s[i-1]), 1
        if r >= utf8.RuneSelf {
550            r, size = utf8.DecodeLastRune(s[0:i])
        }
        i -= size
        if f(r) == truth {
            return i
555        }
    }
    return -1
}

595// Runes returns a slice of runes (Unicode code points) equivalent to s.
func Runes(s []byte) []rune {

    i := 0
    for len(s) > 0 {
600        r, l := utf8.DecodeRune(s)
        t[i] = r
        i++
        s = s[l:]
    }
605    return t
}

// Replace returns a copy of the slice s with the first n
// non-overlapping instances of old replaced by new.
610// If old is empty, it matches at the beginning of the slice
// and after each UTF-8 sequence, yielding up to k+1 replacements
// for a k-rune slice.
// If n < 0, there is no limit on the number of replacements.
func Replace(s, old, new []byte, n int) []byte {
615    m := 0
    if n != 0 {
        // Compute number of replacements.
        m = Count(s, old)
    }
620    if m == 0 {
        // Just return a copy.
        return append([]byte(nil), s...)
    }
    if n < 0 || m < n {
625        n = m
    }

    // Apply replacements to buffer.
    t := make([]byte, len(s)+n*(len(new)-len(old)))
630    w := 0
    start := 0
    for i := 0; i < n; i++ {
        j := start
        if len(old) == 0 {
635            if i > 0 {
                _, wid := utf8.DecodeRune(s[start:])
                j += wid
            }
        } else {
640            j += Index(s[start:], old)
        }
        w += copy(t[w:], s[start:j])
        w += copy(t[w:], new)
        start = j + len(old)
645    }
    w += copy(t[w:], s[start:])
    return t[0:w]
}

650// EqualFold reports whether s and t, interpreted as UTF-8 strings,
// are equal under Unicode case-folding.
func EqualFold(s, t []byte) bool {
    for len(s) != 0 && len(t) != 0 {
        // Extract first rune from each.
655        var sr, tr rune
        if s[0] < utf8.RuneSelf {
            sr, s = rune(s[0]), s[1:]
        } else {
            r, size := utf8.DecodeRune(s)
660            sr, s = r, s[size:]
        }
        if t[0] < utf8.RuneSelf {
            tr, t = rune(t[0]), t[1:]
        } else {
665            r, size := utf8.DecodeRune(t)
            tr, t = r, t[size:]
        }

        // If they match, keep going; if not, return false.
670
        // Easy case.
        if tr == sr {
            continue
        }
675
        // Make sr < tr to simplify what follows.
        if tr < sr {
            tr, sr = sr, tr
        }
680        // Fast check for ASCII.
        if tr < utf8.RuneSelf && 'A' <= sr && sr <= 'Z' {
            // ASCII, and sr is upper case.  tr must be lower case.
            if tr == sr+'a'-'A' {
                continue
685            }
            return false
        }

        // General case.  SimpleFold(x) returns the next equivalent rune > x
690        // or wraps around to smaller values.
        r := unicode.SimpleFold(sr)
        for r != sr && r < tr {
            r = unicode.SimpleFold(r)
        }
695        if r == tr {
            continue
        }
        return false
    }
700
    // One string is empty.  Are both?
    return len(s) == len(t)
}

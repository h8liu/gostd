        }
        if tt.Dump == "" {

310// dumpRegexp writes an encoding of the syntax tree for the regexp re to b.
// It is used during testing to distinguish between parses that might print
// the same using re's String method.
func dumpRegexp(b *bytes.Buffer, re *Regexp) {
    if int(re.Op) >= len(opNames) || opNames[re.Op] == "" {
315        fmt.Fprintf(b, "op%d", re.Op)
    } else {
        switch re.Op {
        default:
            b.WriteString(opNames[re.Op])
320        case OpStar, OpPlus, OpQuest, OpRepeat:
            if re.Flags&NonGreedy != 0 {
                b.WriteByte('n')
            }
            b.WriteString(opNames[re.Op])
325        case OpLiteral:
            if len(re.Rune) > 1 {
                b.WriteString("str")
            } else {
                b.WriteString("lit")
330            }
            if re.Flags&FoldCase != 0 {
                for _, r := range re.Rune {
                    if unicode.SimpleFold(r) != r {
                        b.WriteString("fold")
335                        break
                    }
                }
            }
        }
340    }
    b.WriteByte('{')
    switch re.Op {
    case OpEndText:
        if re.Flags&WasDollar == 0 {
345            b.WriteString(`\z`)
        }
    case OpLiteral:
        for _, r := range re.Rune {
            b.WriteRune(r)
350        }
    case OpConcat, OpAlternate:
        for _, sub := range re.Sub {
            dumpRegexp(b, sub)
        }
355    case OpStar, OpPlus, OpQuest:
        dumpRegexp(b, re.Sub[0])
    case OpRepeat:
        fmt.Fprintf(b, "%d,%d ", re.Min, re.Max)
        dumpRegexp(b, re.Sub[0])
360    case OpCapture:
        if re.Name != "" {
            b.WriteString(re.Name)
            b.WriteByte(':')
        }
365        dumpRegexp(b, re.Sub[0])
    case OpCharClass:
        sep := ""
        for i := 0; i < len(re.Rune); i += 2 {
            b.WriteString(sep)
370            sep = " "
            lo, hi := re.Rune[i], re.Rune[i+1]
            if lo == hi {
                fmt.Fprintf(b, "%#x", lo)
            } else {
375                fmt.Fprintf(b, "%#x-%#x", lo, hi)
            }
        }
    }
    b.WriteByte('}')
380}

    lo := rune(-1)
385    for i := rune(0); i <= unicode.MaxRune; i++ {
        if f(i) {
            if lo < 0 {
                lo = i
            }
390        } else {
            if lo >= 0 {
                re.Rune = append(re.Rune, lo, i-1)
                lo = -1
            }
395        }
    }
    if lo >= 0 {
        re.Rune = append(re.Rune, lo, unicode.MaxRune)
    }
400    return dump(re)
}

func isUpperFold(r rune) bool {
    if unicode.IsUpper(r) {

        }
        c = unicode.SimpleFold(c)

    return false
415}

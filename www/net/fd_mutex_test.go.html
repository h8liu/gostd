                t.Error("broken")

    }

95func TestMutexPanic(t *testing.T) {
    ensurePanics := func(f func()) {
        defer func() {
            if recover() == nil {
                t.Fatal("does not panic")
100            }
        }()
        f()
    }

105    var mu fdMutex
    ensurePanics(func() { mu.Decref() })
    ensurePanics(func() { mu.RWUnlock(true) })
    ensurePanics(func() { mu.RWUnlock(false) })

110    ensurePanics(func() { mu.Incref(); mu.Decref(); mu.Decref() })
    ensurePanics(func() { mu.RWLock(true); mu.RWUnlock(true); mu.RWUnlock(true) })
    ensurePanics(func() { mu.RWLock(false); mu.RWUnlock(false); mu.RWUnlock(false) })

    // ensure that it's still not broken
115    mu.Incref()
    mu.Decref()
    mu.RWLock(true)
    mu.RWUnlock(true)
    mu.RWLock(false)
120    mu.RWUnlock(false)
}

func TestMutexStress(t *testing.T) {
    P := 8
125    N := int(1e6)
    if testing.Short() {
        P = 4
        N = 1e4
    }
130    defer runtime.GOMAXPROCS(runtime.GOMAXPROCS(P))
    done := make(chan bool)
    var mu fdMutex
    var readState [2]uint64
    var writeState [2]uint64
135    for p := 0; p < P; p++ {
        go func() {
            r := rand.New(rand.NewSource(rand.Int63()))
            for i := 0; i < N; i++ {
                switch r.Intn(3) {
140                case 0:
                    if !mu.Incref() {
                        t.Error("broken")
                        return
                    }
145                    if mu.Decref() {
                        t.Error("broken")
                        return
                    }
                case 1:
150                    if !mu.RWLock(true) {
                        t.Error("broken")
                        return
                    }
                    // Ensure that it provides mutual exclusion for readers.
155                    if readState[0] != readState[1] {
                        t.Error("broken")
                        return
                    }
                    readState[0]++
160                    readState[1]++
                    if mu.RWUnlock(true) {
                        t.Error("broken")
                        return
                    }
165                case 2:
                    if !mu.RWLock(false) {
                        t.Error("broken")
                        return
                    }
170                    // Ensure that it provides mutual exclusion for writers.
                    if writeState[0] != writeState[1] {
                        t.Error("broken")
                        return
                    }
175                    writeState[0]++
                    writeState[1]++
                    if mu.RWUnlock(false) {
                        t.Error("broken")
                        return
180                    }
                }
            }
            done <- true
        }()
185    }
    for p := 0; p < P; p++ {
        <-done
    }
    if !mu.IncrefAndClose() {
190        t.Fatal("broken")
    }
    if !mu.Decref() {
        t.Fatal("broken")
    }
195}

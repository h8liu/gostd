    // Consider the following situation.
    // First M calls runtimepark() in runtimeselectgo() passing the sel.
    // Once runtimepark() has unlocked the last lock, another M makes

    // When the G runs on another M, it locks all the locks and frees sel.

    }
515    if cas.receivedp != nil {
        *cas.receivedp = true
    }
    if cas.elem != nil {
        memmove(cas.elem, sg.elem, uintptr(c.elemsize))
520    }
    sg.elem = nil
    gp = sg.g
    gp.param = unsafe.Pointer(sg)
    if sg.releasetime != 0 {
525        sg.releasetime = cputicks()
    }
    goready(gp)
    goto retc

530rclose:
    // read at end of closed channel
    selunlock(sel)
    if cas.receivedp != nil {
        *cas.receivedp = false
535    }
    if cas.elem != nil {
        memclr(cas.elem, uintptr(c.elemsize))
    }
    if raceenabled {
540        raceacquire(unsafe.Pointer(c))
    }
    goto retc

syncsend:
545    // can send to sleeping receiver (sg)
    if raceenabled {
        raceReadObjectPC(c.elemtype, cas.elem, cas.pc, chansendpc)
        racesync(c, sg)
    }
550    selunlock(sel)
    if debugSelect {
        print("syncsend: sel=", sel, " c=", c, "\n")
    }
    if sg.elem != nil {
555        memmove(sg.elem, cas.elem, uintptr(c.elemsize))
    }
    sg.elem = nil
    gp = sg.g
    gp.param = unsafe.Pointer(sg)
560    if sg.releasetime != 0 {
        sg.releasetime = cputicks()
    }
    goready(gp)

565retc:
    if cas.releasetime > 0 {
        blockevent(cas.releasetime-t0, 2)
    }
    return cas.pc, cas.so
570
sclose:
    // send on closed channel
    selunlock(sel)
    panic("send on closed channel")
575}

// A runtimeSelect is a single case passed to rselect.
// This must match ../reflect/value.go:/runtimeSelect
585type runtimeSelect struct {
    dir selectDir
    typ unsafe.Pointer // channel type (not used here)
    ch  *hchan         // channel
    val unsafe.Pointer // ptr to data (SendDir) or ptr to receive buffer (RecvDir)
590}

// These values must match ../reflect/value.go:/SelectDir.
type selectDir int

595const (
    _             selectDir = iota
    selectSend              // case Chan <- Send
    selectRecv              // case <-Chan:
    selectDefault           // default
600)

func reflect_rselect(cases []runtimeSelect) (chosen int, recvOK bool) {
    // flagNoScan is safe here, because all objects are also referenced from cases.
    size := selectsize(uintptr(len(cases)))
605    sel := (*_select)(mallocgc(size, nil, flagNoScan))
    newselect(sel, int64(size), int32(len(cases)))
    r := new(bool)
    for i := range cases {
        rc := &cases[i]
610        switch rc.dir {
        case selectDefault:
            selectdefaultImpl(sel, uintptr(i), 0)
        case selectSend:
            if rc.ch == nil {
615                break
            }
            selectsendImpl(sel, rc.ch, uintptr(i), rc.val, 0)
        case selectRecv:
            if rc.ch == nil {
620                break
            }
            selectrecvImpl(sel, rc.ch, uintptr(i), rc.val, r, 0)
        }
    }
625
    pc, _ := selectgoImpl(sel)
    chosen = int(pc)
    recvOK = *r
    return
630}

635    for {

// checksum is a slow but simple implementation of the Adler-32 checksum.
// It is a straight port of the sample code in RFC 1950 section 9.
func checksum(p []byte) uint32 {
    s1, s2 := uint32(1), uint32(0)
65    for _, x := range p {
        s1 = (s1 + uint32(x)) % mod
        s2 = (s2 + s1) % mod
    }
    return s2<<16 | s1
70}

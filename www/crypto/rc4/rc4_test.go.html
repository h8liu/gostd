func testEncrypt(t *testing.T, desc string, c *Cipher, src, expect []byte) {
    dst := make([]byte, len(src))
    c.XORKeyStream(dst, src)
80    for i, v := range dst {
        if v != expect[i] {
            t.Fatalf("%s: mismatch at byte %d:\nhave %x\nwant %x", desc, i, dst, expect)
        }
    }
85}

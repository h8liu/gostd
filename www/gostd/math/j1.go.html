//              j1(0) = 0
//              j1(inf) = 0

// Method -- y1(x):

//      2. For x<2.
//         Since
55//              y1(x) = 2/pi*(j1(x)*(ln(x/2)+Euler)-1/x-x/2+5/64*x**3-...)

60//              U(z) = U0[0] + U0[1]*z + ... + U0[4]*z**4

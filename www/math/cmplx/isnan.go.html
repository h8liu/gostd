// IsNaN returns true if either real(x) or imag(x) is NaN
10// and neither is an infinity.
func IsNaN(x complex128) bool {
    switch {
    case math.IsInf(real(x), 0) || math.IsInf(imag(x), 0):
        return false
15    case math.IsNaN(real(x)) || math.IsNaN(imag(x)):
        return true
    }
    return false
}

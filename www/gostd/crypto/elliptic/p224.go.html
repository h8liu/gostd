// Field element functions.

// The field that we're dealing with is ℤ/pℤ where p = 2**224 - 2**96 + 1.
//
130// Field elements are represented by a FieldElement, which is a typedef to an
// array of 8 uint32's. The value of a FieldElement, a, is:
//   a[0] + 2**28·a[1] + 2**56·a[1] + ... + 2**196·a[7]
//
// Using 28-bit limbs means that there's only 4 bits of headroom, which is less
135// than we would really like. But it has the useful feature that we hit 2**224
// exactly, making the reflections during a reduce much nicer.

    // true.
    top4AllOnes := uint32(0xffffffff)
450    for i := 4; i < 8; i++ {
        top4AllOnes &= out[i]
    }
    top4AllOnes |= 0xf0000000
    // Now we replicate any zero bits to all the bits in top4AllOnes.
455    top4AllOnes &= top4AllOnes >> 16
    top4AllOnes &= top4AllOnes >> 8
    top4AllOnes &= top4AllOnes >> 4
    top4AllOnes &= top4AllOnes >> 2
    top4AllOnes &= top4AllOnes >> 1
460    top4AllOnes = uint32(int32(top4AllOnes<<31) >> 31)

    // Now we test whether the bottom three limbs are non-zero.
    bottom3NonZero := out[0] | out[1] | out[2]
    bottom3NonZero |= bottom3NonZero >> 16
465    bottom3NonZero |= bottom3NonZero >> 8
    bottom3NonZero |= bottom3NonZero >> 4
    bottom3NonZero |= bottom3NonZero >> 2
    bottom3NonZero |= bottom3NonZero >> 1
    bottom3NonZero = uint32(int32(bottom3NonZero<<31) >> 31)
470
    // Everything depends on the value of out[3].
    //    If it's > 0xffff000 and top4AllOnes != 0 then the whole value is >= p
    //    If it's = 0xffff000 and top4AllOnes != 0 and bottom3NonZero != 0,
    //      then the whole value is >= p
475    //    If it's < 0xffff000, then the whole value is < p
    n := out[3] - 0xffff000
    out3Equal := n
    out3Equal |= out3Equal >> 16
    out3Equal |= out3Equal >> 8
480    out3Equal |= out3Equal >> 4
    out3Equal |= out3Equal >> 2
    out3Equal |= out3Equal >> 1
    out3Equal = ^uint32(int32(out3Equal<<31) >> 31)

485    // If out[3] > 0xffff000 then n's MSB will be zero.
    out3GT := ^uint32(int32(n) >> 31)

    mask := top4AllOnes & ((out3Equal & bottom3NonZero) | out3GT)
    out[0] -= 1 & mask
490    out[3] -= 0xffff000 & mask
    out[4] -= 0xfffffff & mask
    out[5] -= 0xfffffff & mask
    out[6] -= 0xfffffff & mask
    out[7] -= 0xfffffff & mask
495}

// Group element functions.
//
// These functions deal with group elements. The group is an elliptic curve
500// group with a = -3 defined in FIPS 186-3, section D.2.2.

// p224AddJacobian computes *out = a+b where a != b.
func p224AddJacobian(x3, y3, z3, x1, y1, z1, x2, y2, z2 *p224FieldElement) {
    // See http://hyperelliptic.org/EFD/g1p/auto-shortw-jacobian-3.html#addition-p224Add-2007-bl
505    var z1z1, z2z2, u1, u2, s1, s2, h, i, j, r, v p224FieldElement
    var c p224LargeFieldElement

    z1IsZero := p224IsZero(z1)
    z2IsZero := p224IsZero(z2)
510
    // Z1Z1 = Z1²
    p224Square(&z1z1, z1, &c)
    // Z2Z2 = Z2²
    p224Square(&z2z2, z2, &c)
515    // U1 = X1*Z2Z2
    p224Mul(&u1, x1, &z2z2, &c)
    // U2 = X2*Z1Z1
    p224Mul(&u2, x2, &z1z1, &c)
    // S1 = Y1*Z2*Z2Z2
520    p224Mul(&s1, z2, &z2z2, &c)
    p224Mul(&s1, y1, &s1, &c)
    // S2 = Y2*Z1*Z1Z1
    p224Mul(&s2, z1, &z1z1, &c)
    p224Mul(&s2, y2, &s2, &c)
525    // H = U2-U1
    p224Sub(&h, &u2, &u1)
    p224Reduce(&h)
    xEqual := p224IsZero(&h)
    // I = (2*H)²
530    for j := 0; j < 8; j++ {
        i[j] = h[j] << 1
    }
    p224Reduce(&i)
    p224Square(&i, &i, &c)
535    // J = H*I
    p224Mul(&j, &h, &i, &c)
    // r = 2*(S2-S1)
    p224Sub(&r, &s2, &s1)
    p224Reduce(&r)
540    yEqual := p224IsZero(&r)
    if xEqual == 1 && yEqual == 1 && z1IsZero == 0 && z2IsZero == 0 {
        p224DoubleJacobian(x3, y3, z3, x1, y1, z1)
        return
    }
545    for i := 0; i < 8; i++ {
        r[i] <<= 1
    }
    p224Reduce(&r)
    // V = U1*I
550    p224Mul(&v, &u1, &i, &c)
    // Z3 = ((Z1+Z2)²-Z1Z1-Z2Z2)*H
    p224Add(&z1z1, &z1z1, &z2z2)
    p224Add(&z2z2, z1, z2)
    p224Reduce(&z2z2)
555    p224Square(&z2z2, &z2z2, &c)
    p224Sub(z3, &z2z2, &z1z1)
    p224Reduce(z3)
    p224Mul(z3, z3, &h, &c)
    // X3 = r²-J-2*V
560    for i := 0; i < 8; i++ {
        z1z1[i] = v[i] << 1
    }
    p224Add(&z1z1, &j, &z1z1)
    p224Reduce(&z1z1)
565    p224Square(x3, &r, &c)
    p224Sub(x3, x3, &z1z1)
    p224Reduce(x3)
    // Y3 = r*(V-X3)-2*S1*J
    for i := 0; i < 8; i++ {
570        s1[i] <<= 1
    }
    p224Mul(&s1, &s1, &j, &c)
    p224Sub(&z1z1, &v, x3)
    p224Reduce(&z1z1)
575    p224Mul(&z1z1, &z1z1, &r, &c)
    p224Sub(y3, &z1z1, &s1)
    p224Reduce(y3)

    p224CopyConditional(x3, x2, z1IsZero)
580    p224CopyConditional(x3, x1, z2IsZero)
    p224CopyConditional(y3, y2, z1IsZero)
    p224CopyConditional(y3, y1, z2IsZero)
    p224CopyConditional(z3, z2, z1IsZero)
    p224CopyConditional(z3, z1, z2IsZero)
585}

// p224DoubleJacobian computes *out = a+a.
func p224DoubleJacobian(x3, y3, z3, x1, y1, z1 *p224FieldElement) {
    var delta, gamma, beta, alpha, t p224FieldElement
590    var c p224LargeFieldElement

    p224Square(&delta, z1, &c)
    p224Square(&gamma, y1, &c)
    p224Mul(&beta, x1, &gamma, &c)
595
    // alpha = 3*(X1-delta)*(X1+delta)
    p224Add(&t, x1, &delta)
    for i := 0; i < 8; i++ {
        t[i] += t[i] << 1
600    }
    p224Reduce(&t)
    p224Sub(&alpha, x1, &delta)
    p224Reduce(&alpha)
    p224Mul(&alpha, &alpha, &t, &c)
605
    // Z3 = (Y1+Z1)²-gamma-delta
    p224Add(z3, y1, z1)
    p224Reduce(z3)
    p224Square(z3, z3, &c)
610    p224Sub(z3, z3, &gamma)
    p224Reduce(z3)
    p224Sub(z3, z3, &delta)
    p224Reduce(z3)

615    // X3 = alpha²-8*beta
    for i := 0; i < 8; i++ {
        delta[i] = beta[i] << 3
    }
    p224Reduce(&delta)
620    p224Square(x3, &alpha, &c)
    p224Sub(x3, x3, &delta)
    p224Reduce(x3)

    // Y3 = alpha*(4*beta-X3)-8*gamma²
625    for i := 0; i < 8; i++ {
        beta[i] <<= 2
    }
    p224Sub(&beta, &beta, x3)
    p224Reduce(&beta)
630    p224Square(&gamma, &gamma, &c)
    for i := 0; i < 8; i++ {
        gamma[i] <<= 3
    }
    p224Reduce(&gamma)
635    p224Mul(y3, &alpha, &beta, &c)
    p224Sub(y3, y3, &gamma)
    p224Reduce(y3)
}

640// p224CopyConditional sets *out = *in iff the least-significant-bit of control
// is true, and it runs in constant time.
func p224CopyConditional(out, in *p224FieldElement, control uint32) {
    control <<= 31
    control = uint32(int32(control) >> 31)
645
    for i := 0; i < 8; i++ {
        out[i] ^= (out[i] ^ in[i]) & control
    }
}

// p224ToAffine converts from Jacobian to affine form.
func p224ToAffine(x, y, z *p224FieldElement) (*big.Int, *big.Int) {
    var zinv, zinvsq, outx, outy p224FieldElement
    var tmp p224LargeFieldElement
675
    if isPointAtInfinity := p224IsZero(z); isPointAtInfinity == 1 {
        return new(big.Int), new(big.Int)
    }

680    p224Invert(&zinv, z)
    p224Square(&zinvsq, &zinv, &tmp)
    p224Mul(x, x, &zinvsq, &tmp)
    p224Mul(&zinvsq, &zinvsq, &zinv, &tmp)
    p224Mul(y, y, &zinvsq, &tmp)
685
    p224Contract(&outx, x)
    p224Contract(&outy, y)
    return p224ToBig(&outx), p224ToBig(&outy)
}

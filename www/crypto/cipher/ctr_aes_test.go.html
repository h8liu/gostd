        if err != nil {
            t.Errorf("%s: NewCipher(%d bytes) = %s", test, len(tt.key), err)
75            continue

        for j := 0; j <= 5; j += 5 {

80            ctr := cipher.NewCTR(c, tt.iv)

                t.Errorf("%s/%d: CTR\ninpt %x\nhave %x\nwant %x", test, len(in), in, encrypted, out)

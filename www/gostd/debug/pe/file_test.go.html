func isOptHdrEq(a, b interface{}) bool {
    switch va := a.(type) {
    case *OptionalHeader32:
175        vb, ok := b.(*OptionalHeader32)
        if !ok {
            return false
        }
        return *vb == *va
180    case *OptionalHeader64:
        vb, ok := b.(*OptionalHeader64)
        if !ok {
            return false
        }
185        return *vb == *va
    case nil:
        return b == nil
    }
    return false
190}

// See http://www.hyperelliptic.org/EFD/g1p/auto-shortw.html
type Curve interface {

    IsOnCurve(x, y *big.Int) bool

    Double(x1, y1 *big.Int) (x, y *big.Int)
    // ScalarMult returns k*(Bx,By) where k is a number in big-endian form.

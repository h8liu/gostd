15// It's used by sha1block_generic.go and tests.
func blockGeneric(dig *digest, p []byte) {
    var w [16]uint32

    h0, h1, h2, h3, h4 := dig.h[0], dig.h[1], dig.h[2], dig.h[3], dig.h[4]
20    for len(p) >= chunk {
        // Can interlace the computation of w with the
        // rounds below if needed for speed.
        for i := 0; i < 16; i++ {
            j := i * 4
25            w[i] = uint32(p[j])<<24 | uint32(p[j+1])<<16 | uint32(p[j+2])<<8 | uint32(p[j+3])
        }

        a, b, c, d, e := h0, h1, h2, h3, h4

30        // Each of the four 20-iteration rounds
        // differs only in the computation of f and
        // the choice of K (_K0, _K1, etc).
        i := 0
        for ; i < 16; i++ {
35            f := b&c | (^b)&d
            a5 := a<<5 | a>>(32-5)
            b30 := b<<30 | b>>(32-30)
            t := a5 + f + e + w[i&0xf] + _K0
            a, b, c, d, e = t, a, b30, c, d
40        }
        for ; i < 20; i++ {
            tmp := w[(i-3)&0xf] ^ w[(i-8)&0xf] ^ w[(i-14)&0xf] ^ w[(i)&0xf]

45            f := b&c | (^b)&d
            a5 := a<<5 | a>>(32-5)
            b30 := b<<30 | b>>(32-30)
            t := a5 + f + e + w[i&0xf] + _K0
            a, b, c, d, e = t, a, b30, c, d
50        }
        for ; i < 40; i++ {

            w[i&0xf] = tmp<<1 | tmp>>(32-1)
            f := b ^ c ^ d
55            a5 := a<<5 | a>>(32-5)
            b30 := b<<30 | b>>(32-30)
            t := a5 + f + e + w[i&0xf] + _K1
            a, b, c, d, e = t, a, b30, c, d
        }
60        for ; i < 60; i++ {

            w[i&0xf] = tmp<<1 | tmp>>(32-1)
            f := ((b | c) & d) | (b & c)

65            a5 := a<<5 | a>>(32-5)
            b30 := b<<30 | b>>(32-30)
            t := a5 + f + e + w[i&0xf] + _K2
            a, b, c, d, e = t, a, b30, c, d
        }
70        for ; i < 80; i++ {

            w[i&0xf] = tmp<<1 | tmp>>(32-1)
            f := b ^ c ^ d
            a5 := a<<5 | a>>(32-5)
75            b30 := b<<30 | b>>(32-30)
            t := a5 + f + e + w[i&0xf] + _K3

        }

80        h0 += a

        h3 += d

85

    }

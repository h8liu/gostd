335// isMacro reports whether p is a package dependency macro
// (uppercase name).
func isMacro(p string) bool {
    return 'A' <= p[0] && p[0] <= 'Z'
}
340
func allowed(pkg string) map[string]bool {
    m := map[string]bool{}
    var allow func(string)
    allow = func(p string) {
345        if m[p] {
            return
        }
        m[p] = true // set even for macros, to avoid loop on cycle

350        // Upper-case names are macro-expanded.
        if isMacro(p) {
            for _, pp := range pkgDeps[p] {
                allow(pp)
            }
355        }
    }
    for _, pp := range pkgDeps[pkg] {
        allow(pp)
    }
360    return m

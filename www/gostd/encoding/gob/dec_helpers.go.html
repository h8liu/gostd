        if n < 0 || uint64(n) != u || n > state.b.Len() {
            errorf("length of string exceeds input size (%d bytes)", u)
325        }
        if n > state.b.Len() {

        }

        if _, err := state.b.Read(data); err != nil {
            errorf("error decoding string: %s", err)
        }
        slice[i] = string(data)
335    }
    return true
}

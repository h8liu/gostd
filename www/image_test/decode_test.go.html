func withinTolerance(c0, c1 color.Color, tolerance int) bool {
    r0, g0, b0, a0 := c0.RGBA()
    r1, g1, b1, a1 := c1.RGBA()
    r := delta(r0, r1)
70    g := delta(g0, g1)
    b := delta(b0, b1)
    a := delta(a0, a1)
    return r <= tolerance && g <= tolerance && b <= tolerance && a <= tolerance
}

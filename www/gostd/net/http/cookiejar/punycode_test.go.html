func TestPunycode(t *testing.T) {
    for _, tc := range punycodeTestCases {
155        if got, err := encode("", tc.s); err != nil {
            t.Errorf(`encode("", %q): %v`, tc.s, err)
        } else if got != tc.encoded {
            t.Errorf(`encode("", %q): got %q, want %q`, tc.s, got, tc.encoded)
        }
160    }
}

15func ExampleEncoding_EncodeToString() {
    data := []byte("any + old & data")
    str := base64.StdEncoding.EncodeToString(data)
    fmt.Println(str)
    // Output:
20    // YW55ICsgb2xkICYgZGF0YQ==
}

func ExampleEncoding_DecodeString() {
    str := "c29tZSBkYXRhIHdpdGggACBhbmQg77u/"
25    data, err := base64.StdEncoding.DecodeString(str)
    if err != nil {
        fmt.Println("error:", err)
        return
    }
30    fmt.Printf("%q\n", data)
    // Output:
    // "some data with \x00 and \ufeff"
}

35func ExampleNewEncoder() {
    input := []byte("foo\x00bar")
    encoder := base64.NewEncoder(base64.StdEncoding, os.Stdout)
    encoder.Write(input)
    // Must close the encoder when finished to flush any partial blocks.
40    // If you comment out the following line, the last partial block "r"
    // won't be encoded.
    encoder.Close()
    // Output:
    // Zm9vAGJhcg==
45}

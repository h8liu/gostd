func (f *File) applyRelocations386(dst []byte, rels []byte) error {
    // 8 is the size of Rel32.
590    if len(rels)%8 != 0 {

    }

595    if err != nil {
        return err
    }

    b := bytes.NewReader(rels)
600    var rel Rel32

    for b.Len() > 0 {
        binary.Read(b, f.ByteOrder, &rel)
        symNo := rel.Info >> 8
605        t := R_386(rel.Info & 0xff)

        if symNo == 0 || symNo > uint32(len(symbols)) {
            continue

            f.ByteOrder.PutUint32(dst[rel.Off:rel.Off+4], val)
        }
620    }

func (f *File) DWARF() (*dwarf.Data, error) {
675    // There are many other DWARF sections, but these

            return nil, err

695    rela := f.Section(".rela.debug_info")

            return nil, err

        if err != nil {
            return nil, err
        }
705    }

    rel := f.Section(".rel.debug_info")

710        data, err := rel.Data()
        if err != nil {
            return nil, err
        }

715        if err != nil {
            return nil, err
        }
    }

    }
725

730            if err != nil && uint64(len(b)) < s.Size {
                return nil, err
            }

735                if r.Type != SHT_RELA && r.Type != SHT_REL {

                    continue
740                }

                if err != nil {

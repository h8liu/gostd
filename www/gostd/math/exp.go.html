30// exp(x)
// Returns the exponential of x.
//
// Method
//   1. Argument reduction:
35//      Reduce x to an r so that |r| <= 0.5*ln2 ~ 0.34658.
//      Given x, find r and integer k such that
//
//               x = k*ln2 + r,  |r| <= 0.5*ln2.
//
40//      Here r will be represented as r = hi-lo for better
//      accuracy.
//
//   2. Approximation of exp(r) by a special rational function on
//      the interval [0,0.34658]:
45//      Write
//          R(r**2) = r*(exp(r)+1)/(exp(r)-1) = 2 + r*r/6 - r**4/360 + ...
//      We use a special Remes algorithm on [0,0.34658] to generate
//      a polynomial of degree 5 to approximate R. The maximum error
//      of this polynomial approximation is bounded by 2**-59. In
50//      other words,
//          R(z) ~ 2.0 + P1*z + P2*z**2 + P3*z**3 + P4*z**4 + P5*z**5
//      (where z=r*r, and the values of P1 to P5 are listed below)
//      and
//          |                  5          |     -59
55//          | 2.0+P1*z+...+P5*z   -  R(z) | <= 2
//          |                             |
//      The computation of exp(r) thus becomes
//                             2*r
//              exp(r) = 1 + -------
60//                            R - r
//                                 r*R1(r)
//                     = 1 + r + ----------- (for better accuracy)
//                                2 - R1(r)
//      where
65//                               2       4             10
//              R1(r) = r - (P1*r  + P2*r  + ... + P5*r   ).
//
//   3. Scale back to obtain exp(x):
//      From step 1, we have
70//         exp(x) = 2**k * exp(r)
//
// Special cases:
//      exp(INF) is INF, exp(NaN) is NaN;
//      exp(-INF) is 0, and
75//      for finite argument, only exp(0)=1 is exact.
//
// Accuracy:
//      according to an error analysis, the error is always less than
//      1 ulp (unit in the last place).
80//
// Misc. info.
//      For IEEE double
//          if x >  7.09782712893383973096e+02 then exp(x) overflow
//          if x < -7.45133219101941108420e+02 then exp(x) underflow
85//
// Constants:
// The hexadecimal values are the intended ones for the following
// constants. The decimal values may be used, provided that the
// compiler will convert from decimal to binary accurately enough
90// to produce the hexadecimal values shown.

func exp(x float64) float64 {
    const (
        Ln2Hi = 6.93147180369123816490e-01
95        Ln2Lo = 1.90821492927058770002e-10
        Log2e = 1.44269504088896338700e+00

        Overflow  = 7.09782712893383973096e+02
        Underflow = -7.45133219101941108420e+02
100        NearZero  = 1.0 / (1 << 28) // 2**-28
    )

    // special cases
    switch {
105    case IsNaN(x) || IsInf(x, 1):
        return x
    case IsInf(x, -1):
        return 0
    case x > Overflow:
110        return Inf(1)
    case x < Underflow:
        return 0
    case -NearZero < x && x < NearZero:
        return 1 + x
115    }

    // reduce; computed as r = hi - lo for extra precision.
    var k int
    switch {
120    case x < 0:
        k = int(Log2e*x - 0.5)
    case x > 0:
        k = int(Log2e*x + 0.5)
    }
125    hi := x - float64(k)*Ln2Hi
    lo := float64(k) * Ln2Lo

    // compute
    return expmulti(hi, lo, k)
130}
